function showReviewModal(id) { const modal = document.getElementById('reviewModal'); const form = document.getElementById('reviewForm'); form.reset(); document.getElementById('reviewId').value = ''; document.getElementById('reviewImages').innerHTML = ''; document.getElementById('reviewMeta').innerText = ''; fetch('/admin/reviews/api/' + id) .then(function(res) { return res.json(); }) .then(function(data) { if (!data || !data.id || data.success === false) { alert(data && data.message ? data.message : 'Không tải được dữ liệu'); return; } document.getElementById('reviewId').value = data.id; document.getElementById('reviewUserName').value = (data.user_name || '') + (data.user_email ? ' (' + data.user_email + ')' : ''); document.getElementById('reviewPlaceName').value = data.place_name || ('#' + data.destination_id); document.getElementById('reviewRating').value = String(data.rating || 5); document.getElementById('reviewComment').value = data.comment || ''; document.getElementById('reviewMeta').innerText = 'ID #' + data.id + ' · Địa điểm #' + data.destination_id + ' · Người dùng #' + data.user_id; const imagesEl = document.getElementById('reviewImages'); const images = Array.isArray(data.images) ? data.images : []; if (!images.length) { imagesEl.innerHTML = 'Không có ảnh'; } else { imagesEl.innerHTML = images.map(function(url) { return ''; }).join(''); } modal.classList.remove('hidden'); }) .catch(function() { alert('Lỗi mạng'); }); } function closeReviewModal() { document.getElementById('reviewModal').classList.add('hidden'); } async function saveReview() { const form = document.getElementById('reviewForm'); if (!form.checkValidity()) { form.reportValidity(); return; } const formData = new FormData(form); const data = Object.fromEntries(formData.entries()); const res = await fetch('/admin/reviews/save', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(data) }); const body = await res.json().catch(function() { return {}; }); if (res.ok && body.success) { location.reload(); } else { alert(body.message || 'Lỗi khi lưu'); } } async function deleteReview(id) { if (!confirm('Bạn có chắc chắn muốn xóa đánh giá này? Điểm trung bình của địa điểm sẽ được cập nhật lại.')) return; const res = await fetch('/admin/reviews/delete/' + id, { method: 'POST' }); const body = await res.json().catch(function() { return {}; }); if (res.ok && body.success) { location.reload(); } else { alert(body.message || 'Lỗi khi xóa đánh giá'); } }
