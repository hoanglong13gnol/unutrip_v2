data.latitude : ''; document.getElementById('destLng').value = data.longitude != null ? data.longitude : ''; document.getElementById('destOpen').value = data.open_time || ''; document.getElementById('destClose').value = data.close_time || ''; }) .catch(function() { alert('Lỗi mạng'); }); } else { title.innerText = 'Thêm Địa điểm Mới'; } modal.classList.remove('hidden'); } function closeModal() { document.getElementById('destModal').classList.add('hidden'); } async function saveDest() { const form = document.getElementById('destForm'); if (!form.checkValidity()) { form.reportValidity(); return; } const formData = new FormData(form); const data = Object.fromEntries(formData.entries()); const res = await fetch('/admin/destinations/save', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(data) }); const body = await res.json().catch(function() { return {}; }); if (res.ok && body.success !== false) { location.reload(); } else { alert(body.message || body.error || 'Lỗi khi lưu dữ liệu'); } } async function deleteDest(id) { if (!confirm('Xóa địa điểm này khỏi app_places?
